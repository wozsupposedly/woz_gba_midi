/*
  GBA_Midi_RP2040_Woz.txt

  Goal: Stable RP2040-first multiboot uploader for Spritesmods GBA MIDI ROM,
  then USB/DIN MIDI forwarding after the GBA runtime is loaded.

  Key design choices:
  - Direct GPIO register operations for deterministic edge timing.
  - Cycle-based waits (busy_wait_at_least_cycles) instead of microsecond timers.
  - Same tx/rx bit ordering and handshake flow as original gbaserial.c.
  - Embedded gbamidi.gba payload in flash, retried until upload succeeds.
  - USB MIDI and DIN MIDI stay idle until the ROM upload reports success.

  Voltage/domain note:
  - RP2040 is 3.3V logic. Use proper wiring/level handling to the GBA link port.
*/

}

void selectGbaCableProfile(uint8_t profile) {
  activeCableProfile = profile % GBA_CABLE_PROFILE_COUNT;

  // Release both possible SD paths before enabling the active profile.
  pinMode(PIN_GBA_SD_CABLE_A, INPUT);
  pinMode(PIN_GBA_SD_CABLE_B, INPUT);

  gbaPinSC = PIN_GBA_SC_FIXED;
  gbaPinSI = PIN_GBA_SI_FIXED;
  if (activeCableProfile == 0) {
    gbaPinSD = PIN_GBA_SD_CABLE_A;
  } else {
    gbaPinSD = PIN_GBA_SD_CABLE_B;
  }
}

void gbaSerInitPins() {
  pinMode(gbaPinSC, INPUT_PULLUP);
  pinMode(gbaPinSD, INPUT_PULLUP);
  pinMode(gbaPinSI, OUTPUT);
  digitalWrite(gbaPinSI, HIGH);
}

uint16_t gbaSerRxWord() {
  const uint32_t scMask = pinMask(gbaPinSC);
  const uint32_t sdMask = pinMask(gbaPinSD);
  const uint32_t siMask = pinMask(gbaPinSI);

  uint16_t ret = 0;
  // SD input with pull-up enabled.
  gpioOutDisableFast(sdMask);
  gpio_pull_up(gbaPinSD);

  waitCycles(GBA_CHAR_CYCLES);

  // Tell GBA we want a word by lowering SI.
  gpioSetLowFast(siMask);

  // Wait for start bit.
  uint32_t startWaitUs = time_us_32();
  do {
    while (gpioReadAllFast() & sdMask) {
      if (timeoutExpired(startWaitUs, GBA_WORD_TIMEOUT_US)) {
        gpioOutEnableFast(scMask);
        gpioSetHighFast(scMask);
        gpioSetHighFast(siMask);
        return 0xFFFF;
      }
    }
    waitCycles(GBA_CHAR_CYCLES / 2);
  } while (gpioReadAllFast() & sdMask);

  waitCycles(GBA_CHAR_CYCLES);  // start bit

  for (uint8_t x = 0; x < 16; x++) {
    ret >>= 1;
    if (gpioReadAllFast() & sdMask) {
      ret |= 0x8000;
    }
    waitCycles(GBA_CHAR_CYCLES);
  }

  waitCycles(GBA_CHAR_CYCLES);  // stop bit

  // Done receiving.
  gpioSetHighFast(siMask);
  gpioOutEnableFast(scMask);
  gpioSetHighFast(scMask);

  return ret;
}

void gbaSerTxWord(uint16_t data) {
  const uint32_t scMask = pinMask(gbaPinSC);
  const uint32_t sdMask = pinMask(gbaPinSD);
  const uint32_t siMask = pinMask(gbaPinSI);

  // SD and SC input with pull-ups.
  gpioOutDisableFast(sdMask);
  gpioOutDisableFast(scMask);
  gpio_pull_up(gbaPinSD);
  gpio_pull_up(gbaPinSC);

  // Tell GBA we're sending.
  gpioSetHighFast(siMask);

  // Wait for SD and SC high.
  uint32_t startWaitUs = time_us_32();
  while ((gpioReadAllFast() & (scMask | sdMask)) != (scMask | sdMask)) {
    waitCycles(GBA_CHAR_CYCLES / 2);
    if (timeoutExpired(startWaitUs, GBA_WORD_TIMEOUT_US)) {
      return;
    }
  }

  // Wait 8 bit-times, then drive SD and SC low.
  waitCycles(GBA_CHAR_CYCLES * 8);
  gpioOutEnableFast(scMask | sdMask);
  gpioSetLowFast(scMask | sdMask);

  waitCycles(GBA_CHAR_CYCLES);  // start bit

  for (uint8_t x = 0; x < 16; x++) {
    if (data & 1U) {
      gpioSetHighFast(sdMask);
    } else {
      gpioSetLowFast(sdMask);
    }
    waitCycles(GBA_CHAR_CYCLES);
    data >>= 1;
  }

  gpioSetHighFast(sdMask);
  waitCycles(GBA_CHAR_CYCLES);  // stop bit
}

uint16_t gbaSerXferWord(uint16_t data) {
  noInterrupts();
  gbaSerTxWord(data);
  uint16_t r = gbaSerRxWord();
  interrupts();
  return r;
}

// ---------------- CRC/encryption (from original) ----------------
static uint32_t gbaEncryptSeed = 0;
static uint32_t gbaCrcVal = 0;
static uint8_t gbaCrcVar1 = 0;
static uint16_t gbaCrcVar8 = 0;

void gbaEncryptSetSeed(uint8_t data) {
  gbaEncryptSeed = (static_cast<uint32_t>(data) << 8) | 0xFFFF00C1UL;
}

uint32_t gbaEncryptWord(uint32_t data, uint32_t offset) {
  gbaEncryptSeed = (gbaEncryptSeed * 0x6F646573UL) + 1UL;
  uint32_t pos = ~(offset + 0x02000000UL) + 1UL;
  return (data ^ gbaEncryptSeed) ^ (pos ^ 0x6465646FUL);
}

void gbaCrcInit(uint8_t var1, uint16_t var8) {
  gbaCrcVal = 0xFFF8UL;
  gbaCrcVar1 = var1;
  gbaCrcVar8 = var8;
}

void gbaCrcAdd(uint32_t data) {
  for (uint8_t bit = 0; bit < 32; bit++) {
    uint32_t var30 = gbaCrcVal ^ data;
    gbaCrcVal >>= 1;
    data >>= 1;
    if (var30 & 1UL) {
      gbaCrcVal ^= 0x0A517UL;
    }
  }
}

uint16_t gbaCrcFinalize(uint32_t data) {
  uint32_t crctemp = ((((data & 0xFF00UL) + gbaCrcVar8) << 8) | 0xFFFF0000UL) + gbaCrcVar1;
  gbaCrcAdd(crctemp);
  return static_cast<uint16_t>(gbaCrcVal & 0xFFFFUL);
}

uint32_t gbaAppChecksumWord(const uint8_t* rom, uint32_t imageLength) {
  uint32_t sum = 0;
  for (uint32_t pos = 0xE0; pos + 3 < imageLength; pos += 4) {
    uint32_t word = static_cast<uint32_t>(rom[pos + 0]);
    word |= static_cast<uint32_t>(rom[pos + 1]) << 8;
    word |= static_cast<uint32_t>(rom[pos + 2]) << 16;
    word |= static_cast<uint32_t>(rom[pos + 3]) << 24;
    sum += word;
  }
  return 0UL - sum;
}

uint8_t gbaUploadByte(const uint8_t* rom, uint32_t imageLength, uint32_t index, uint32_t checksumWord) {
  if (index < imageLength) {
    return rom[index];
  }
  index -= imageLength;
  if (index < 4) {
    return static_cast<uint8_t>((checksumWord >> (index * 8U)) & 0xFFU);
  }
  return 0xFF;
}

bool gbaSendRom(const uint8_t* rom, uint32_t length) {
  if (length == 0) {
    return false;
  }

  uint32_t uploadLength = GBA_ORIGINAL_SEND_LEN;
  uint32_t checksumWord = gbaAppChecksumWord(rom, length);
  uint32_t romsize = (uploadLength & 0xFFFFFFF0UL) + 16UL;

  uint16_t rd = 0;
  uint8_t timeout = 30;
  // Wait for GBA handshake.
  do {
    delay(100);
    rd = gbaSerXferWord(0x6202);
    timeout--;
  } while (rd != 0x7202 && timeout > 0);

  if (rd != 0x7202) {
    return false;
  }

  // Send 0xC0 header bytes (unencrypted)
  gbaSerXferWord(0x6100);
  uint32_t ptr = 0;
  for (uint8_t i = 0; i < 0x60; i++) {
    uint16_t d = static_cast<uint16_t>(gbaUploadByte(rom, length, ptr++, checksumWord));
    d |= static_cast<uint16_t>(gbaUploadByte(rom, length, ptr++, checksumWord)) << 8;
    gbaSerXferWord(d);
  }

  // Seed exchange
  gbaSerXferWord(0x6202);
  gbaSerXferWord(0x63C1);
  rd = gbaSerXferWord(0x63C1);
  gbaEncryptSetSeed(static_cast<uint8_t>(rd & 0xFF));

  uint8_t var1 = static_cast<uint8_t>(((rd & 0xFF) + 0x20F) & 0xFF);
  uint16_t conf = static_cast<uint16_t>(var1) | 0x6400U;
  gbaSerXferWord(conf);

  // Length + CRC init
  uint16_t len = static_cast<uint16_t>(((romsize - 0xC0UL) >> 2) - 0x34UL);
  uint16_t var8 = gbaSerXferWord(len);
  gbaCrcInit(var1, var8);

  // Send encrypted payload
  for (uint32_t pos = 0xC0; pos < romsize; pos += 4) {
    uint32_t d = static_cast<uint32_t>(gbaUploadByte(rom, length, pos + 0, checksumWord));
    d |= static_cast<uint32_t>(gbaUploadByte(rom, length, pos + 1, checksumWord)) << 8;
    d |= static_cast<uint32_t>(gbaUploadByte(rom, length, pos + 2, checksumWord)) << 16;
    d |= static_cast<uint32_t>(gbaUploadByte(rom, length, pos + 3, checksumWord)) << 24;

    gbaCrcAdd(d);
    d = gbaEncryptWord(d, pos);

    gbaSerXferWord(static_cast<uint16_t>(d & 0xFFFFUL));
    gbaSerXferWord(static_cast<uint16_t>((d >> 16) & 0xFFFFUL));
  }

  // Finish + CRC exchange
  gbaSerXferWord(0x0065);
  gbaSerXferWord(0x0065);

  timeout = 30;
  while (gbaSerXferWord(0x0065) != 0x0075 && timeout > 0) {
    delay(100);
    timeout--;
  }
  if (timeout == 0) {
    return false;
  }

  uint16_t gbaCrc = gbaSerXferWord(0x0066);
  uint16_t localCrc = gbaCrcFinalize(gbaCrc);
  gbaSerXferWord(localCrc);

  return true;
}

void serviceMultiboot() {
  if (gbaRuntimeReady) {
    return;
  }

  uint32_t now = millis();
  if (lastAttemptMs != 0 && (now - lastAttemptMs) < MULTIBOOT_RETRY_MS) {
    return;
  }
  lastAttemptMs = now;

  blinkAttempt();
  selectGbaCableProfile(nextCableProfile);
  gbaSerInitPins();

  if (gbaSendRom(gbaRomImage, GBA_ROM_IMAGE_LEN)) {
    gbaRuntimeReady = true;
    blinkSuccess();
    delay(3000);  // same settle delay as original
    return;
  }

  cableAttemptsOnThisProfile++;
  if (cableAttemptsOnThisProfile >= GBA_CABLE_ATTEMPTS_BEFORE_SWITCH) {
    cableAttemptsOnThisProfile = 0;
    nextCableProfile = (nextCableProfile + 1) % GBA_CABLE_PROFILE_COUNT;
  }
}

void forwardMidiByteToGba(uint8_t b) {
  if (!gbaRuntimeReady) {
    return;
  }

  // Spritesmods runtime consumes one MIDI byte in the low byte of a 16-bit link word.
  gbaSerTxWord(static_cast<uint16_t>(b));
}

void serviceUsbMidi() {
  while (usb_midi.available() > 0) {
    int v = usb_midi.read();
    if (v < 0) {
      break;
    }

    uint8_t b = static_cast<uint8_t>(v);
    forwardMidiByteToGba(b);
    if (USB_TO_DIN_THRU) {
      Serial1.write(b);
    }
  }
}

void serviceDinMidi() {
  while (Serial1.available() > 0) {
    uint8_t b = static_cast<uint8_t>(Serial1.read());
    forwardMidiByteToGba(b);
    if (DIN_TO_USB_THRU) {
      usb_midi.write(b);
    }
  }
}

void setup() {
  pinMode(PIN_STATUS_LED, OUTPUT);
  digitalWrite(PIN_STATUS_LED, LOW);

  if (!TinyUSBDevice.isInitialized()) {
    TinyUSBDevice.begin(0);
  }
  usb_midi.setStringDescriptor("GBA MIDI Woz");
  usb_midi.begin();
  if (TinyUSBDevice.mounted()) {
    TinyUSBDevice.detach();
    delay(10);
    TinyUSBDevice.attach();
  }

  Serial1.setTX(PIN_DIN_MIDI_TX);
  Serial1.setRX(PIN_DIN_MIDI_RX);
  Serial1.begin(DIN_MIDI_BAUD);

  selectGbaCableProfile(nextCableProfile);
  gbaSerInitPins();

  // Let GBA fully boot to BIOS logo before first attempt.
  delay(2000);
}

void loop() {
#ifdef TINYUSB_NEED_POLLING_TASK
  TinyUSBDevice.task();
#endif

  serviceMultiboot();
  serviceUsbMidi();
  serviceDinMidi();
}
